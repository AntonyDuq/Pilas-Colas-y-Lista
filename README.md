def ordenar-insercion(lista):
   for indice in rango(1,len(lista)):
       valor = lista[indice]
       i = indice - 1
       while i<=0.
          if valor < lista[i]:
             lista[i+1] = lista[i]

             lista[i] = valor 
             i = i - 1
         else:
              break
